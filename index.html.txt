    <script>
        const API_BASE = 'https://trendhire.onrender.com';
        
        async function testTrendingJobs() {
            showLoading();
            try {
                const response = await fetch(`${API_BASE}/trending-jobs`);
                const data = await response.json();
                showResults('Trending Jobs Data', JSON.stringify(data, null, 2));
                updateChart(data.trending_jobs);
            } catch (error) {
                showResults('Error', 'Failed to fetch trending jobs: ' + error.message);
            }
        }
        
        async function testSkillsAnalysis() {
            showLoading();
            try {
                const response = await fetch(`${API_BASE}/skills-analysis/Python,React,SQL`);
                const data = await response.json();
                showResults('Skills Analysis Result', JSON.stringify(data, null, 2));
            } catch (error) {
                showResults('Error', 'Failed to analyze skills: ' + error.message);
            }
        }
        
        async function testHealthCheck() {
            showLoading();
            try {
                const response = await fetch(`${API_BASE}/health`);
                const data = await response.json();
                showResults('Health Check', JSON.stringify(data, null, 2));
            } catch (error) {
                showResults('Error', 'Failed to check health: ' + error.message);
            }
        }
        
        function showLoading() {
            document.getElementById('loading').style.display = 'block';
            document.getElementById('apiResults').style.display = 'none';
        }
        
        function showResults(title, content) {
            document.getElementById('loading').style.display = 'none';
            const resultsDiv = document.getElementById('apiResults');
            resultsDiv.innerHTML = `<h4>${title}</h4><pre>${content}</pre>`;
            resultsDiv.style.display = 'block';
        }
        
        function updateChart(jobData) {
            const ctx = document.getElementById('trendChart').getContext('2d');
            new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: jobData.map(job => job.title),
                    datasets: [{
                        label: 'Growth Rate (%)',
                        data: jobData.map(job => job.growth),
                        backgroundColor: [
                            'rgba(102, 126, 234, 0.8)',
                            'rgba(118, 75, 162, 0.8)',
                            'rgba(255, 99, 132, 0.8)',
                            'rgba(54, 162, 235, 0.8)',
                            'rgba(255, 206, 86, 0.8)'
                        ],
                        borderColor: [
                            'rgba(102, 126, 234, 1)',
                            'rgba(118, 75, 162, 1)',
                            'rgba(255, 99, 132, 1)',
                            'rgba(54, 162, 235, 1)',
                            'rgba(255, 206, 86, 1)'
                        ],
                        borderWidth: 2
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        y: {
                            beginAtZero: true,
                            title: {
                                display: true,
                                text: 'Growth Rate (%)'
                            }
                        },
                        x: {
                            title: {
                                display: true,
                                text: 'Job Titles'
                            }
                        }
                    },
                    plugins: {
                        title: {
                            display: true,
                            text: 'AI Job Market Growth Trends'
                        }
                    }
                }
            });
        }
        
        // Initialize chart with default data
        window.onload = function() {
            testTrendingJobs();
        };
